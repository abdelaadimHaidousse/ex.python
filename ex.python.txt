def SubString(word,c): 

    sub = ""
    for i in range(len( word)):
        if c ==  word[i]:
            sub = word[0:i+1]
    return sub   
def Stars(StarsNumber):
    s=""
    for i in range(StarsNumber):
        s += "*"
    return s
